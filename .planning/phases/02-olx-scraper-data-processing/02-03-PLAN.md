1. Create `lib/scrapers/olx/olx-phone.ts`:
   ```typescript
   import { chromium } from 'playwright-extra'
   import StealthPlugin from 'puppeteer-extra-plugin-stealth'
   import { OLX_SELECTORS } from './olx-selectors'

   chromium.use(StealthPlugin())

   export async function revealPhone(listingUrl: string): Promise<string | null> {
     const browser = await chromium.launch({ headless: true })
     try {
       const page = await browser.newPage()
       await page.goto(listingUrl, { waitUntil: 'domcontentloaded', timeout: 15000 })
       const phoneBtn = page.locator(OLX_SELECTORS.phoneButton)
       if (await phoneBtn.count() > 0) {
         await phoneBtn.first().click()
         await page.waitForTimeout(1500)
         const phoneEl = page.locator(OLX_SELECTORS.phoneNumber)
         if (await phoneEl.count() > 0) {
           const phoneText = await phoneEl.first().textContent()
           return phoneText?.trim() ?? null
         }
       }
       return null
     } catch (err) {
       console.error(`[olx-phone] Failed to reveal phone for ${listingUrl}:`, err)
       return null
     } finally {
       await browser.close()
     }
   }
   ```

2. Create `lib/scrapers/olx/olx-scraper.ts` — the main OlxScraper class:
   ```typescript
   import got from 'got'
   import type { ScraperAdapter, ScraperConfig, RawLead } from '../types'
   import { buildListingUrl } from './olx-urls'
   import { parseListingIndex, parseListingDetail } from './olx-parser'
   import { revealPhone } from './olx-phone'
   import pLimit from 'p-limit'

   export async function delayWithJitter(baseMs: number, jitterMs: number): Promise<void> {
     const delay = baseMs + Math.random() * jitterMs
     await new Promise(resolve => setTimeout(resolve, delay))
   }

   const httpClient = got.extend({
     headers: {
       'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/124.0.0.0 Safari/537.36',
       'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
       'Accept-Language': 'pl-PL,pl;q=0.9,en-US;q=0.8,en;q=0.7',
       'Accept-Encoding': 'gzip, deflate, br',
     },
     retry: { limit: 2, statusCodes: [429, 503] },
     timeout: { request: 15000 },
     responseType: 'text',
   })

   export class OlxScraper implements ScraperAdapter {
     name = 'olx'
     private config: ScraperConfig

     constructor(config: ScraperConfig) {
       this.config = config
     }

     async run(): Promise<RawLead[]> {
       const leads: RawLead[] = []
       const limit = pLimit(this.config.concurrency || 1)

       for (const category of this.config.categories) {
         for (const city of this.config.cities) {
           for (const keyword of this.config.keywords.length ? this.config.keywords : ['']) {
             for (let page = 1; page <= this.config.maxPages; page++) {
               await delayWithJitter(this.config.delayMs, this.config.jitterMs)
               const url = buildListingUrl(category, city, keyword, page)
               try {
                 const { body, statusCode } = await httpClient.get(url)
                 if (statusCode === 403) {
                   console.error(`[olx-scraper] 403 Forbidden at ${url} — stopping`)
                   return leads
                 }
                 const listings = parseListingIndex(body)
                 if (listings.length === 0) break // no more listings on this page

                 const detailPromises = listings.map(listing =>
                   limit(async () => {
                     await delayWithJitter(this.config.delayMs, this.config.jitterMs)
                     try {
                       const { body: detailBody } = await httpClient.get(listing.url)
                       const partial = parseListingDetail(detailBody)
                       const phone = await revealPhone(listing.url)
                       const rawLead: RawLead = {
                         sourceUrl: listing.url,
                         sourcePlatform: 'olx',
                         name: partial.name ?? null,
                         phone: phone,
                         email: null, // OLX does not expose email
                         city: partial.city ?? null,
                         description: partial.description ?? null,
                         categories: partial.categories ?? [],
                         priceMin: partial.priceMin ?? null,
                         priceMax: partial.priceMax ?? null,
                         socialLinks: partial.socialLinks ?? {},
                         sellerType: partial.sellerType ?? 'unknown',
                         listingCount: partial.listingCount ?? null,
                         scrapedAt: new Date().toISOString(),
                       }
                       leads.push(rawLead)
                     } catch (err) {
                       console.error(`[olx-scraper] Failed to scrape detail ${listing.url}:`, err)
                     }
                   })
                 )
                 await Promise.all(detailPromises)
               } catch (err) {
                 console.error(`[olx-scraper] Failed to fetch index ${url}:`, err)
               }
             }
           }
         }
       }
       return leads
     }
   }
   ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| OLX.pl HTTP responses -> got client | Untrusted remote HTML; could contain malicious content |
| OLX.pl JS execution -> Playwright | Phone reveal runs untrusted JS in sandboxed browser |
| Parsed HTML fields -> RawLead | Raw strings from HTML go into application data structures |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-06 | Tampering | lib/scrapers/olx/olx-parser.ts | mitigate | Cheerio parses HTML safely (no JS execution); extracted text is stored as raw strings; Zod validation in types.ts rejects structurally invalid data before DB write |
| T-02-07 | Denial of Service | lib/scrapers/olx/olx-scraper.ts | mitigate | p-limit(1) prevents concurrent browser contexts; configurable delay+jitter prevents request bursts; 403 detection stops the job immediately |
| T-02-08 | Denial of Service | lib/scrapers/olx/olx-phone.ts | mitigate | try/finally ensures browser.close() always runs; 15s page timeout prevents hanging; single browser instance per call with p-limit gate |
| T-02-09 | Information Disclosure | lib/scrapers/olx/olx-scraper.ts User-Agent | accept | Static UA header mimics Chrome browser; standard practice for web scraping; no PII leaked |
